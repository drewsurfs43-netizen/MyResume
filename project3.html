About the Project

This project creates a graphical scene using Python’s Turtle graphics module. The program uses shapes such as circles, rectangles, and polygons to create a complete scene including a sun, clouds, a house, grass, and other elements. The file below contains the exact code used to draw the entire scene.

Project Code


'''Drew Olsen'''

from turtle import *
# variable to switch between day and night
day_night = "day"
if day_night == "day":
    bgcolor("lightblue")
else:
    bgcolor("black")
'''this is a sun'''
# move the turtle to a new location
penup()
goto (200,200)
pendown()
# make a circle
fillcolor("yellow")
begin_fill()
circle(50)
end_fill()

# Draw a house
color("black")  
penup()
goto(-150, -200)
pendown()

# House base
fillcolor("blue")
begin_fill()
for _ in range(4):
    forward(200)
    left(90)
end_fill()

# Roof
penup()
goto(-150, 0)
pendown()
fillcolor("brown")
begin_fill()
left(45)
forward(140)
right(90)
forward(140)
end_fill()

# Door
penup()
goto(-70, -200)
pendown()
fillcolor("white")
begin_fill()
for _ in range(2):
    forward(50)
    left(90)
    forward(100)
    left(90)
end_fill()

# Grass
penup()
goto(-300, -250)
pendown()
color("green")
fillcolor("green")
begin_fill()
for _ in range(2):
    forward(600)
    left(90)
    forward(50)
    left(90)
end_fill()

# Cloud 1
penup()
goto(-200, 150)
pendown()
color("white")
fillcolor("white")
begin_fill()
circle(40)
end_fill()

penup()
goto(-160, 150)
pendown()
begin_fill()
circle(40)
end_fill()

penup()
goto(-120, 150)
pendown()
begin_fill()
circle(40)
end_fill()

# Cloud 2
penup()
goto(100, 150)
pendown()
begin_fill()
circle(40)
end_fill()

penup()
goto(140, 150)
pendown()
begin_fill()
circle(40)
end_fill()

penup()
goto(180, 150)
pendown()
begin_fill()
circle(40)
end_fill()

done()
